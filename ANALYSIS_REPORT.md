### 4. SQL 쿼리 (데이터베이스 연동 시)
```sql
-- 가용율 낮은 상품 (P0 우선순위)
SELECT productCode, productName, availableQuantity, quantity
FROM inventory
WHERE availabilityRate <= 30
ORDER BY availabilityRate ASC;

-- 미판매 상품 (P1 우선순위)
SELECT productCode, productName, daysWithoutOrder, totalStockValue
FROM inventory
WHERE daysWithoutOrder >= 100
ORDER BY daysWithoutOrder DESC;

-- 카테고리별 수익 분석
SELECT category, SUM(totalStockValue) as totalValue, 
       AVG(marginPercentage) as avgMargin
FROM inventory
GROUP BY category
ORDER BY totalValue DESC;
```
